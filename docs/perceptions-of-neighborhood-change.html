chgmeans_tbl <- with(dcas16svy, 
                 svyby(~nhdchg, ~raceeth, svymean, na.rm=TRUE)) %>%

        raceeth = relevel(factor(
            rep(c("White", "Asian", "Black", "Latino"), 3)), ref="White"),
